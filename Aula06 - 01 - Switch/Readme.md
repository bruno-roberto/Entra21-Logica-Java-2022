# Condicionais Switch

## Exemplo de uso 


	opcoes = entrada.nextLine();
		switch(opcoes) {
		case "verde":
			System.out.println("A cor da esperança!");
			break;
		case "azul":
			System.out.println("A cor do céu!");
			break;
		case "amarelo":
			System.out.println("Cor do Ouro!");
			break;
		case "vermelho":
			System.out.println("A cor do sangue!");
			break;
		default:
			System.out.println("Escolha uma opção valida.");
			break;
		}
